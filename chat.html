    <script>

        const chatMessages = document.getElementById('chatMessages');

        const messageInput = document.getElementById('messageInput');

        const sendButton = document.getElementById('sendButton');

        const typingIndicator = document.getElementById('typingIndicator');


        // Przykładowe odpowiedzi AI (w przyszłości można podłączyć prawdziwe API)

        const aiResponses = [

            "Rozumiem, że to może być trudne. Chcesz mi opowiedzieć więcej o tym, co Cię niepokoi?",

            "To brzmi jak bardzo stresująca sytuacja. Jak długo się z tym zmagasz?",

            "Twoje uczucia są całkowicie zrozumiałe. Każdy czasami przechodzi przez trudne momenty.",

            "Dziękuję, że się ze mną dzielisz. To wymaga odwagi, aby mówić o swoich problemach.",

            "Pamiętaj, że nie jesteś sam/sama w tym. Czy próbowałeś/aś jakichś technik relaksacyjnych?",

            "To, co opisujesz, to normalna reakcja na stres. Czy chciałbyś/abyś poznać kilka technik radzenia sobie?",

            "Słyszę, że to dla Ciebie ważne. Jak myślisz, co mogłoby Ci teraz pomóc?",

            "Twoje uczucia są ważne i zasługują na uwagę. Czy jest coś konkretnego, co chciałbyś/abyś zmienić?",

            "Rozumiem Twój punkt widzenia. Czasami pomocne jest spojrzenie na sytuację z innej perspektywy.",

            "To brzmi jak duże wyzwanie. Pamiętaj, że małe kroki też prowadzą do celu."

        ];


        function addMessage(message, isUser = false) {

            const messageDiv = document.createElement('div');

            messageDiv.className = `message ${isUser ? 'user-message' : 'ai-message'}`;

            messageDiv.textContent = message;

            chatMessages.appendChild(messageDiv);

            chatMessages.scrollTop = chatMessages.scrollHeight;

        }


        function showTypingIndicator() {

            typingIndicator.style.display = 'block';

            chatMessages.scrollTop = chatMessages.scrollHeight;

        }


        function hideTypingIndicator() {

            typingIndicator.style.display = 'none';

        }


        function getAIResponse() {

            return aiResponses[Math.floor(Math.random() * aiResponses.length)];

        }


        function sendMessage() {

            const message = messageInput.value.trim();

            if (!message) return;


            // Dodaj wiadomość użytkownika

            addMessage(message, true);

            messageInput.value = '';

            

            // Wyłącz przycisk i pokaż wskaźnik pisania

            sendButton.disabled = true;

            showTypingIndicator();


            // Symuluj opóźnienie odpowiedzi AI

            setTimeout(() => {

                hideTypingIndicator();

                const aiResponse = getAIResponse();

                addMessage(aiResponse, false);

                sendButton.disabled = false;

                messageInput.focus();

            }, 1500 + Math.random() * 1000); // 1.5-2.5 sekundy

        }


        function handleKeyPress(event) {

            if (event.key === 'Enter') {

                sendMessage();

            }

        }


        function goBack() {

            window.location.href = 'index.html';

        }


        // Fokus na input po załadowaniu strony

        window.onload = function() {

            messageInput.focus();

        };
